### 思路：采用二维数组方式实现
### 核心：当前数据=上一列的数+上一列前一个数

## 一、输出一个10行10列的图案
* 思路：定义一个二维数组，并给数组赋值为0；

```
#include <stdio.h>

/**
 * 杨辉三角
 */
int main(){

    int max =10;

    int arr[max][max];

    for (int x = 0; x < max; ++x) {
        for (int y = 0; y < max; ++y) {
            arr[x][y] =0;
            printf(" %d ",arr[x][y]);
        }
        printf("\n");
    }

    return 0;
}
```

## 三、添加杨辉三角算法
* 上一行前一列+上一行当前列=当前数
```
for (int x = 0; x < max; ++x) {                         //行
        for (int y = 0; y < max; ++y) {                     //列
            arr[x][y] =0;
            if (x == y | y == 0){
                arr[x][y] =1;
            } else {
                arr[x][y] =arr[x-1][y-1] +arr[x-1][y];      //上一行前一列+上一行当前列
            }
            printf(" %d ",arr[x][y]);
        }
        printf("\n");
    }
```

 ## 六、最终版
 ```
 #include <stdio.h>

/**
 * 杨辉三角
 */
int main(){

    int max =10;
    int arr[max][max];

    //数据初始化
    for (int x = 0; x < max; ++x) {                         //行
        for (int y = 0 ; y <= x; ++y) {                     //列

            if (x == y | y == 0){
                arr[x][y] =1;
            } else {
                arr[x][y] =arr[x-1][y-1] +arr[x-1][y];      //上一行前一列+上一行当前列
            }
        }
    }

    //数据按格式输出
    for (int i = 0; i < 10; ++i) {                          //第一列打印
        printf("%*d",20-i*2,arr[i][0]);
        for (int j=1;j<=i;j++){
            printf("%4d",arr[i][j]);
        }
        printf("\n");
    }

    return 0;
}
 ```
